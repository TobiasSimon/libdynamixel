cmake_minimum_required(VERSION 2.6)
PROJECT(LIBDYNAMIXEL)


SET(LIBDYNAMIXEL_VERSION_MAJOR 0)
SET(LIBDYNAMIXEL_VERSION_MINOR 0)
SET(LIBDYNAMIXEL_VERSION_MICRO 1)

INCLUDE (CheckIncludeFiles)
CHECK_INCLUDE_FILES("sys/ioctl.h" HAVE_TIOCSRS485)
CHECK_INCLUDE_FILES("sys/ioctl.h" HAVE_TIOCM_RTS)

CHECK_INCLUDE_FILES("termios.h" B1000000)
CONFIGURE_FILE(${CMAKE_CURRENT_SOURCE_DIR}/config.h.in ${CMAKE_CURRENT_BINARY_DIR}/config.h)


MESSAGE(STATUS "libdynamixel C_FLAGS: ${CMAKE_C_FLAGS}")

ADD_LIBRARY(dynamixel dynamixel.c dynamixel-rtu.c)

SET(CMAKE_CXX_FLAGS "-DHAVE_POSIXSERIAL -g -Wall")
