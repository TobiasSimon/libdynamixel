cmake_minimum_required(VERSION 2.6)
project(LIBDYNAMIXEL)


set(LIBDYNAMIXEL_VERSION_MAJOR 0)

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -DLIBDYNAMIXEL_VERSION_MICRO=${LIBDYNAMIXEL_VERSION_MICRO}")


message(STATUS "libdynamixel C_FLAGS: ${CMAKE_C_FLAGS}")

add_library (libdynamixel dynamixel.c dynamixel-rtu.c)
